Google Apps Scriptの設定手順：

  1. Google Sheetsを開き「拡張機能」→「Apps Script」
  2. 既存のコードを消去し、下記のコード（doGet付き）を貼り付けて保存
  3. 「デプロイ」→「新しいデプロイ」→「ウェブアプリ」
  4. 「アクセスできるユーザー」を「全員」に設定してデプロイ
  5. 発行されたURLをここに貼り付け
function doGet(e) {
  let ss;
  try {
    ss = e.parameter.id ? SpreadsheetApp.openById(e.parameter.id) : SpreadsheetApp.getActiveSpreadsheet();
  } catch(err) {
    ss = SpreadsheetApp.getActiveSpreadsheet();
  }
  const sheet = ss.getSheetByName(e.parameter.sheet || 'Sheet1');
  let data = [];
  if (sheet.getLastRow() > 0) {
    data = sheet.getDataRange().getValues();
  }
  return ContentService
    .createTextOutput(JSON.stringify({result:'ok', data: data}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  let ss;
  try {
    ss = e.parameter.id ? SpreadsheetApp.openById(e.parameter.id) : SpreadsheetApp.getActiveSpreadsheet();
  } catch(err) {
    ss = SpreadsheetApp.getActiveSpreadsheet();
  }
  const sheet = ss.getSheetByName(e.parameter.sheet || 'Sheet1');
  const data = JSON.parse(e.parameter.data);
  sheet.appendRow(data);
  return ContentService
    .createTextOutput(JSON.stringify({result:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}
